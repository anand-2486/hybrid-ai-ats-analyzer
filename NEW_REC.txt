"use client";

import React, { useState } from "react";
import SkillsMatrix from "./SkillsMatrix";
import { FileText, Star, ShieldCheck, AlertTriangle, Fingerprint, Search, Sparkles, UserCheck, Target, Lightbulb, Code2, Milestone, ChevronRight, CheckSquare } from "lucide-react";

export default function RecruiterDashboard({ candidatesList }) {
  const [selectedIdx, setSelectedIdx] = useState(0);
  const activeCandidate = candidatesList[selectedIdx] || null;

  if (!activeCandidate) return <p className="text-sm text-slate-500">No applicants parsed.</p>;

  const formatCategoryLabel = (str) => {
    return str.replace(/_/g, " ").replace(/\b\w/g, (char) => char.toUpperCase());
  };

  const renderFormattedBullet = (bulletText, highlightClass = "text-emerald-400") => {
    let cleanText = bulletText.replace(/^[\s·•\-*]+/, "").trim();
    const headingMatch = cleanText.match(/^(\*\*|)(.*?)\1\s*:\s*(.*)$/s);

    if (headingMatch) {
      let heading = headingMatch[2].trim();
      let body = headingMatch[3].trim();
      heading = heading.replace(/\*\*$/, "").replace(/:$/, "").trim();

      return (
        <span className="text-xs text-slate-300 leading-relaxed">
          <strong className={`font-bold uppercase tracking-wide text-[11px] block md:inline mb-1 md:mb-0 ${highlightClass}`}>
            {heading}:
          </strong>{" "}
          {body}
        </span>
      );
    }
    return <span className="text-xs text-slate-300 leading-relaxed">{cleanText}</span>;
  };

  const calculateGradeMetric = (score) => {
    if (score >= 90) return { label: "GRADE: A+", style: "bg-emerald-500/5 text-emerald-400 border-emerald-500/20" };
    if (score >= 80) return { label: "GRADE: A-", style: "bg-green-500/5 text-green-400 border-green-500/20" };
    if (score >= 70) return { label: "GRADE: B-", style: "bg-emerald-500/5 text-emerald-400 border-emerald-500/20" };
    if (score >= 60) return { label: "GRADE: C+", style: "bg-amber-500/5 text-amber-400 border-amber-500/20" };

  };

  const rawTargetRole = activeCandidate.target_role || activeCandidate.ats_details?.target_role || "Frontend Developer Intern";
  const isMLOrDataTrack = rawTargetRole.toLowerCase().includes("ml") || 
                          rawTargetRole.toLowerCase().includes("machine") || 
                          rawTargetRole.toLowerCase().includes("data");

  // ── 🌟 ROLE-BASED CONDITIONAL ROADMAP FACTORY ──

    {
      phase: "Phase 1: Pure Algorithm Design",
      title: "Implement Mathematical Logic From Scratch",
      stack: "Python, NumPy, Math Vector Matrix Calculus",
      action: "Avoid standard black-box libraries like scikit-learn. Write probabilistic logic (e.g., Naive Bayes text token classification) manually to prove pure engineering depth."

    {
      phase: "Phase 2: Data Pipeline Engineering",
      title: "Ingestion & Sanitization Architectures",
      stack: "Pandas, Vector Database Indices, Tokenization Arrays",
      action: "Build manual structural formatting pipelines to handle missing data variables, tracking dimensional inconsistencies with math execution logs."

      phase: "Phase 3: Model Evaluation Tuning",
      title: "Custom Performance Profiler Suites",
      stack: "Confusion Matrices, Precise Loss Derivatives, Metrics Validation",
      action: "Construct evaluation models from absolute zero constraints to audit precision, recall variance, and prediction weight tolerances manually."

    {
      phase: "Phase 4: High-Performance Production Ingestion",
      title: "Deploy Live Vector Classification Pipelines",
      stack: "FastAPI, Docker Containment, Asynchronous Inference Workers",
      action: "Package the custom vectorized calculation algorithms into performant background execution threads serving cross-origin application interfaces smoothly."

    {
      phase: "Phase 1: Typings & Structural Rigor",
      title: "Transition to Strict TypeScript",
      stack: "TypeScript, Strict Linting, Component Structural Contracts",
      action: "Refactor core loose javascript state configurations to strict interface contracts to eliminate component runtime data parsing failures."

    {
      phase: "Phase 2: Live Concurrency State",
      title: "Real-time Sync & State Architectures",
      stack: "WebSockets, Socket.io, Context API, Zustand Management",
      action: "Construct an active canvas board grid or dynamic tracking pane handling background state synchronizations fluidly across connections."
    },
    {
      phase: "Phase 3: Algorithmic Full-Stack Logic",
      title: "Custom Service Network Implementation",
      stack: "Node.js, Express Layer, Manual Routing Protocols",
      action: "Develop persistent routing utilities and custom validation blocks from scratch to capture deep problem-solving metrics."
    },
    {
      phase: "Phase 4: Enterprise Production Deployment",
      title: "Optimize Layout Renders & Pipelines",

                    MATCH
                  </span>

                    {candidate.match_index || candidate.ats_score || 0}%

          </div>
          <div className="bg-[#0B0F17]/40 p-4 border border-[#1E2638]/60 rounded-xl flex items-center justify-between shadow-inner w-full gap-4 mt-2">

      {/* ── SECTION 7: DYNAMIC AI CAREER ROADMAP PROGRESSION PANEL ── */}
      <div className="w-full bg-[#111622]/40 border border-[#1E2638] rounded-2xl p-6 shadow-xl shadow-black/10 backdrop-blur-md space-y-5">
        
        <div className="flex flex-col sm:flex-row sm:items-center justify-between border-b border-[#1E2638]/60 pb-3.5 gap-3">

          ))}
        </div>
